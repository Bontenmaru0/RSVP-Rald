// Insert guest record
CREATE OR REPLACE FUNCTION insert_invitation_passcode(
    p_passcode VARCHAR(6),
    p_name VARCHAR(50) DEFAULT NULL,
    p_guests INTEGER DEFAULT NULL
)
RETURNS invitation_passcodes
LANGUAGE plpgsql
AS $$
DECLARE
    v_row invitation_passcodes;
BEGIN
    -- Passcode cannot be null or empty
    IF p_passcode IS NULL OR trim(p_passcode) = '' THEN
        RAISE EXCEPTION 'Passcode cannot be null or empty.';
    END IF;

    -- Passcode must be exactly 6 characters
    IF char_length(trim(p_passcode)) <> 6 THEN
        RAISE EXCEPTION 'Passcode must be exactly 6 characters.';
    END IF;

    -- Passcode must be unique
    IF EXISTS (
        SELECT 1
        FROM invitation_passcodes
        WHERE passcode = p_passcode
    ) THEN
        RAISE EXCEPTION 'Passcode "%" already exists.', p_passcode;
    END IF;

    INSERT INTO invitation_passcodes (
        passcode,
        name,
        guests
    )
    VALUES (
        trim(p_passcode),
        p_name,
        p_guests
    )
    RETURNING * INTO v_row;

    RETURN v_row;
END;
$$;

//Get guests info by Passcode
CREATE OR REPLACE FUNCTION get_invitation_by_passcode(
    p_passcode VARCHAR(6)
)
RETURNS invitation_passcodes
LANGUAGE plpgsql
AS $$
DECLARE
    v_row invitation_passcodes;
BEGIN
    -- Validate passcode
    IF p_passcode IS NULL OR trim(p_passcode) = '' THEN
        RAISE EXCEPTION 'Passcode cannot be null or empty.';
    END IF;

    IF char_length(trim(p_passcode)) <> 6 THEN
        RAISE EXCEPTION 'Passcode must be exactly 6 characters.';
    END IF;

    -- Get invitation
    SELECT *
    INTO v_row
    FROM invitation_passcodes
    WHERE passcode = trim(p_passcode);

    -- Not found
    IF NOT FOUND THEN
        RAISE EXCEPTION 'Invalid passcode.';
    END IF;

    RETURN v_row;
END;
$$;

// admin check
CREATE OR REPLACE FUNCTION is_admin_passcode(
    p_passcode VARCHAR(6)
)
RETURNS BOOLEAN
LANGUAGE sql
AS $$
    SELECT EXISTS (
        SELECT 1
        FROM invitation_passcodes
        WHERE passcode = trim(p_passcode)
          AND admin = TRUE
    );
$$;

//get guest list
CREATE OR REPLACE FUNCTION admin_get_guests(
    p_passcode VARCHAR(6) DEFAULT NULL,
    p_name VARCHAR(50) DEFAULT NULL,
    p_guests INTEGER DEFAULT NULL,
    p_confirmation_status VARCHAR(20) DEFAULT NULL,
    p_datetime_sent DATE DEFAULT NULL,
    p_datetime_updated_by_admin DATE DEFAULT NULL,
    p_sort_direction VARCHAR(4) DEFAULT 'ASC'
)
RETURNS TABLE (
    id BIGINT,
    passcode VARCHAR(6),
    name VARCHAR(50),
    guests INTEGER,
    datetime_sent TIMESTAMPTZ,
    datetime_updated_by_admin TIMESTAMPTZ,
    confirmation_status VARCHAR(20)
)
LANGUAGE plpgsql
AS $$
BEGIN
    -- Validate sort direction
    IF UPPER(p_sort_direction) NOT IN ('ASC', 'DESC') THEN
        p_sort_direction := 'ASC';
    END IF;

    RETURN QUERY EXECUTE format(
        '
        SELECT
            i.id,
            i.passcode,
            i.name,
            i.guests,
            i.datetime_sent,
            i.datetime_updated_by_admin,
            i.confirmation_status
        FROM invitation_passcodes i
        WHERE
            ($1 IS NULL OR i.passcode ILIKE ''%%'' || trim($1) || ''%%'')
        AND ($2 IS NULL OR i.name ILIKE ''%%'' || trim($2) || ''%%'')
        AND ($3 IS NULL OR i.guests = $3)
        AND ($4 IS NULL OR i.confirmation_status = $4)
        AND ($5 IS NULL OR i.datetime_sent::DATE = $5)
        AND ($6 IS NULL OR i.datetime_updated_by_admin::DATE = $6)
        ORDER BY i.id %s',
        UPPER(p_sort_direction)
    )
    USING
        p_passcode,
        p_name,
        p_guests,
        p_confirmation_status,
        p_datetime_sent,
        p_datetime_updated_by_admin;
END;
$$;

//update guests status
CREATE OR REPLACE FUNCTION admin_update_confirmation_status(
    p_passcode VARCHAR(6),
    p_status VARCHAR(20)
)
RETURNS TEXT
LANGUAGE plpgsql
AS $$
DECLARE
    v_name VARCHAR(50);
BEGIN
    -- Validate status
    IF p_status NOT IN ('ForConfirmation', 'Confirmed', 'Declined') THEN
        RAISE EXCEPTION 'Unknown confirmation status.';
    END IF;

    UPDATE invitation_passcodes
    SET confirmation_status = p_status,
        datetime_updated_by_admin = NOW()
    WHERE passcode = trim(p_passcode)
    RETURNING name INTO v_name;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'Guest not found.';
    END IF;

    RETURN format(
        'Successfully updated %s''s status to %s.',
        COALESCE(v_name, 'Guest'),
        p_status
    );

EXCEPTION
    WHEN OTHERS THEN
        RAISE EXCEPTION 'Something went wrong.';
END;
$$;

//update guests count
CREATE OR REPLACE FUNCTION admin_update_guest_count(
    p_passcode VARCHAR(6),
    p_guests INTEGER
)
RETURNS TEXT
LANGUAGE plpgsql
AS $$
DECLARE
    v_name VARCHAR(50);
BEGIN
    IF p_guests IS NULL OR p_guests < 0 THEN
        RAISE EXCEPTION 'Invalid number of guests.';
    END IF;

    UPDATE invitation_passcodes
    SET guests = p_guests,
        datetime_updated_by_admin = NOW()
    WHERE passcode = trim(p_passcode)
    RETURNING name INTO v_name;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'Guest not found.';
    END IF;

    RETURN format(
        'Successfully updated %s''s guest count to %s.',
        COALESCE(v_name, 'Guest'),
        p_guests
    );

EXCEPTION
    WHEN OTHERS THEN
        RAISE EXCEPTION 'Something went wrong.';
END;
$$;

//guests deletion also ready for bulk checklist deletion
CREATE OR REPLACE FUNCTION admin_delete_guests(
    p_passcodes VARCHAR[]
)
RETURNS TEXT
LANGUAGE plpgsql
AS $$
DECLARE
    v_names TEXT;
    v_count INTEGER;
BEGIN
    SELECT string_agg(name, ', ')
    INTO v_names
    FROM invitation_passcodes
    WHERE passcode = ANY(p_passcodes);

    DELETE FROM invitation_passcodes
    WHERE passcode = ANY(p_passcodes);

    GET DIAGNOSTICS v_count = ROW_COUNT;

    IF v_count = 0 THEN
        RAISE EXCEPTION 'No guests were found.';
    END IF;

    RETURN format(
        'Successfully deleted %s guest(s): %s.',
        v_count,
        COALESCE(v_names, '')
    );

EXCEPTION
    WHEN OTHERS THEN
        RAISE EXCEPTION 'Something went wrong.';
END;
$$;

//status count summary/dashboard
CREATE OR REPLACE FUNCTION admin_get_dashboard_summary()
RETURNS TABLE (
    total_invitations BIGINT,
    confirmed_invitations BIGINT,
    declined_invitations BIGINT,
    for_confirmation_invitations BIGINT,

    total_guests BIGINT,
    confirmed_guests BIGINT,
    declined_guests BIGINT,
    for_confirmation_guests BIGINT
)
LANGUAGE sql
AS $$
    SELECT
        -- Invitation counts
        COUNT(*) AS total_invitations,
        COUNT(*) FILTER (
            WHERE confirmation_status = 'Confirmed'
        ) AS confirmed_invitations,
        COUNT(*) FILTER (
            WHERE confirmation_status = 'Declined'
        ) AS declined_invitations,
        COUNT(*) FILTER (
            WHERE confirmation_status = 'ForConfirmation'
        ) AS for_confirmation_invitations,

        -- Guest counts
        COALESCE(SUM(guests), 0) AS total_guests,
        COALESCE(SUM(guests) FILTER (
            WHERE confirmation_status = 'Confirmed'
        ), 0) AS confirmed_guests,
        COALESCE(SUM(guests) FILTER (
            WHERE confirmation_status = 'Declined'
        ), 0) AS declined_guests,
        COALESCE(SUM(guests) FILTER (
            WHERE confirmation_status = 'ForConfirmation'
        ), 0) AS for_confirmation_guests
    FROM invitation_passcodes;
$$;
